# useTokenGating

Hook to provide minting capabilities for Hybrid contracts.

## Usage

```tsx
import { useAccount } from "wagmi"
import { useTokenGating } from "hybrid"
import { Deployments } from "~/.hybrid/client"

const Component = () => {
	const { address } = useAccount()
	const { isLoading, allow, deny } = useTokenGating({ 
		address, 
		contract: Deployments.NFT 
	})

	return (
		<div>
			<p>Allow Access? - {allow}</p>
			<p>Deny Access? - {deny}</p>
		</div>
	)
}
```

## Returns

```json
{
  isLoading: boolean
  address: string
  allow: boolean
  deny: boolean
}
```
