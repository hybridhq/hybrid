# \<MintButton\>

A context-aware generic MintButton for NFT smart contracts.

## Usage

```tsx
import { TotalSupply } from "hybrid"
import { Deployments } from "~/.hybrid/client"

const MyComponent = () => {
	return (
		<p>
			A total of
			<TotalSupply contract={Deployments.NFT} /> NFTs have been minted.
		</p>
	)
}

export default App
```

import { Deployments } from "~/.hybrid/client"
