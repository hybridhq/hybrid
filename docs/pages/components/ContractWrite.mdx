# \<ContractWrite\>

A context-aware generic ContractWrite for NFT smart contracts.

## Usage

```tsx
import { ContractWrite } from "hybrid"
import { Deployments } from "../.hybrid/client"

const MyComponent = () => {
	return (
		<>
			<ContractWrite
				contract={Deployments.NFT}
				prepare={{
					functionName: "mint",
					args: [1]
				}}
				timeout={2000}
			/>
		</>
	)
}

export default MyComponent
```
