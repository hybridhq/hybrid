# useMinting

Hook to provide minting capabilities for Hybrid contracts.

## Usage

```tsx
import { useMinting } from "hybrid"
import { Deployments } from "../.hybrid/client"

const Component = () => {
	const { mint, totalSupply } = useMinting({
		contract: Deployments.NFT
	})

	return (
		<div>
			<button onClick={mint}>Mint Now</button>
			<p>Total Minted: {totalSupply}</p>
		</div>
	)
}
```

## Returns

```json
{
  isMinting: boolean
  isSuccess: boolean
  isError: boolean
  totalSupply: number | undefined
  mint: (amount: number) => Promise<TransactionReceipt | undefined>
}
```
